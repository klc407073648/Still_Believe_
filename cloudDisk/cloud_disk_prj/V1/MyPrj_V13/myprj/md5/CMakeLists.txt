#头文件目录
INCLUDE_DIRECTORIES( ${FASTDFS_INCLUDE} ${FASTCOMMON_INCLUDE} ${HIREDIS_INCLUDE} ${MYSQL_INCLUDE} ${COMMON_INCLUDE}  ./include )

#目录源码路径
SET( MD5_SRC_DIR ./source )

#代码列表
FILE (GLOB MD5_FILES ${MD5_SRC_DIR}/*.c ${COMMON_SRC_DIR}/*.c )

#依赖的LIB
SET(MD5_LIB  ${LIB_FASTDFS} ${LIB_FASTCOMMON}  ${LIB_HIREDIS} ${LIB_FCGI} ${LIB_M} ${LIB_MYSQL} ${LIB_LOG4CPP})

#生成 
ADD_EXECUTABLE( md5 ${MD5_FILES})

#连接生成相关库
TARGET_LINK_LIBRARIES( md5 ${MD5_LIB} ${LOG4CPPTEST_LIB} -lpthread )
