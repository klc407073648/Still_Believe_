#头文件目录
INCLUDE_DIRECTORIES( ${FASTDFS_INCLUDE} ${FASTCOMMON_INCLUDE} ${HIREDIS_INCLUDE} ${MYSQL_INCLUDE} ${COMMON_INCLUDE}  ./include )

#目录源码路径
SET( SHAREFILES_SRC_DIR ./source )

#代码列表
FILE (GLOB SHAREFILES_FILES ${SHAREFILES_SRC_DIR}/*.c ${COMMON_SRC_DIR}/*.c )

#依赖的LIB
SET(SHAREFILES_LIB  ${LIB_FASTDFS} ${LIB_FASTCOMMON}  ${LIB_HIREDIS} ${LIB_FCGI} ${LIB_M} ${LIB_MYSQL} ${LIB_LOG4CPP})

#生成 
ADD_EXECUTABLE( sharefiles ${SHAREFILES_FILES})

#连接生成相关库
TARGET_LINK_LIBRARIES( sharefiles ${SHAREFILES_LIB} ${LOG4CPPTEST_LIB} -lpthread )
