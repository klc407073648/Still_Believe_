#头文件目录
INCLUDE_DIRECTORIES( ${TINYXML_INCLUDE}  ./include )

#目录源码路径
SET( TINYXMLTEST_SRC_DIR ./source )

#代码列表
FILE (GLOB TINYXMLTEST_FILES ${TINYXMLTEST_SRC_DIR}/*.cpp)

#依赖的LIB
SET(TINYXMLTEST_LIB  ${LIB_TINYXML} )

#生成 
ADD_EXECUTABLE( tinyxmlTest ${TINYXMLTEST_FILES})

#连接生成相关库
TARGET_LINK_LIBRARIES( tinyxmlTest ${TINYXMLTEST_LIB} )
